(jQuery) Use min & max & step with fractional values 

var jqueryExampleMinMaxFractional = $("p.jquery-example-min-max-fractional").awesomeRating({
    valueMin: 10,
    valueStep: 1.5,
    valueMax: 20.5,
    valueInitial: 17.5,
    cssBaseSelected     : "fa-star third-selected",
    cssBaseUnselected   : "fa-star-o third-unselected",
    targetSelector      : "span.jquery-example-min-max-fractional",
    allowFractional     : true,
    calculateFractional : function(current, rate) {
        return 1 - ((rate - current) / 1.5);
    }
    });

(jQuery) Use Custom Values with fractional values allowed and complex CSS, including custom icons 

var jqueryExampleComplex = $("p.jquery-example-complex").awesomeRating({
    valueInitial: "CD",
    values: ["A", "B", "C", "D", "E"],
    cssBaseSelected     : " ",
    cssBaseUnselected   : " ",
    cssValuesSelected : [
        "fa-check-square first-selected",
        "fa-heart second-selected",
        "fa-envelope third-selected",
        "fa-comment forth-selected",
        "fa-circle fifth-selected" ],
    cssValuesUnselected: [
        "fa-check-square-o first-unselected",
        "fa-heart-o second-unselected",
        "fa-envelope-o third-unselected",
        "fa-comment-o forth-unselected",
        "fa-circle-o fifth-unselected" ],
    targetSelector: "span.jquery-example-complex",
    allowFractional: true,
    calculateFractional: function(current, rate) {
        if (current.indexOf(rate) === 1) { return 0.5 };
    return 1;
    }
    });
