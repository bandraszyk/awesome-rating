Awesome Rating

JQuery

Default configuration

Rating value: 

$(".jqueryStandard")
    .awesomeRating({
        valueInitial: 4,
        targetSelector: "span.jqueryStandardValue"
    });

Changing values

Rating value: 

$(".jqueryValues")
    .awesomeRating({
        values: ["A", "B", "C", "D", "E", "F", "G"],
        valueInitial: "D",
        targetSelector: "span.jqueryValuesValue"
    });

CSS adjustments

Rating value: 

$(".jqueryCSS")
    .awesomeRating({
        cssBaseSelected     : "fa-check",
        cssBaseUnselected   : "fa-minus",
        cssValuesSelected : [
            "first-selected",
            "second-selected",
            "third-selected",
            "forth-selected",
            "fifth-selected" ],
        cssValuesUnselected: [
            "first-unselected",
            "second-unselected",
            "third-unselected",
            "forth-unselected",
            "fifth-unselected" ],
        cssHover: "rating-star-hover-transparent",
        valueInitial: 1,
        targetSelector: "span.jqueryCSSValue"
    });

Fractional behaviour

Rating value: 

$(".jqueryFractional")
    .awesomeRating({
        values: ["A", "B", "C", "D", "E", "F"],
        valueInitial: "CD",
        targetSelector: "span.jqueryFractionalValue",
        allowFractional: true,
        calculateFractional: function(current, rate) {
            if (current.indexOf(rate) === 1) { return 0.4; }
            return 1;
        },
        allowFractional: true,
        readonly: true,
        applyHoverCss: false
    });

AngularJS

Default configuration

Rating value: 

ratingApp.controller("StandardController", [ "$scope", function($scope) {
    $scope.rating = 4;
}]);

Changing values

Rating value: 

ratingApp.controller("ValuesController", [ "$scope", function($scope) {
    $scope.rating = "F";
    $scope.options = {
        values: ["A", "B", "C", "D", "E", "F", "G"]
    };
}]);

Default configuration

var model = {
    rating: ko.observable(2)
};
ko.applyBindings(model, $(".knockoutjsStandard")[0]);

Fractional behaviour

Rating value: 

var model = {
    rating: ko.observable("CD"),
    options: {
        values: ["A", "B", "C", "D", "E", "F"],
        allowFractional: true,
        calculateFractional: function(current, rate) {
            if (current.indexOf(rate) === 1) { return 0.7 };
            return 1;
        },
        allowFractional: true,
        readonly: true,
        applyHoverCss: false
    }
};
ko.applyBindings(model, $(".knockoutjsFractional")[0]);
